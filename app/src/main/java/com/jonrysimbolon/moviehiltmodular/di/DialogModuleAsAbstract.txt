package com.jonrysimbolon.moviehiltmodular.di

import com.jonrysimbolon.core.dialog.CustomDialog
import com.jonrysimbolon.core.dialog.CustomDialogReload
import com.jonrysimbolon.core.dialog.ui.Failure
import com.jonrysimbolon.core.dialog.ui.Loading
import dagger.Binds
import dagger.Module
import dagger.hilt.InstallIn
import dagger.hilt.android.components.ActivityComponent
import dagger.hilt.android.scopes.ActivityScoped

@Qualifier
annotation class LoadingAnnotation

@Qualifier
annotation class FailedAnnotation

@InstallIn(ActivityComponent::class)
@Module
abstract class LoadingModule{

    @ActivityScoped
    @Binds
    abstract fun bindLoading(
        impl: Loading
    ): CustomDialog
}

@InstallIn(ActivityComponent::class)
@Module
abstract class FailedModule{

    @ActivityScoped
    @Binds
    abstract fun bindFailed(
        impl: Failure
    ): CustomDialogReload
    
}
